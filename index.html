关于生成艺术

生成艺术是使用算法和自主系统创作的艺术形式。艺术家编写代码，定义规则和参数，然后让计算机执行这些规则，创造出通常具有不可预测性和独特性的艺术作品。

这个画廊展示了四种不同的生成艺术算法，所有作品都是实时在您的浏览器中生成。每次刷新都会创建全新的、独一无二的艺术作品。

function drawFractal(x, y, len, angle) {
    if (len < 5) return;
    
    let x2 = x + len * cos(angle);
    let y2 = y + len * sin(angle);
    
    line(x, y, x2, y2);
    
    // 递归调用，创造分形
    drawFractal(x2, y2, len * 0.67, angle - PI/6);
    drawFractal(x2, y2, len * 0.67, angle + PI/6);
}
